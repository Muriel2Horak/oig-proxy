from pathlib import Path
import re, json, hashlib

root = Path('/Users/martinhorak/Projects/oig-proxy-worktrees/twin-ack-forensics-verification')
source_path = root / '.sisyphus/evidence/task-2-loki-sample.json'
out_path = root / 'analysis/twin-ack-forensics-verification/mqtt-timeline.jsonl'

line_ts_re = re.compile(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}(?:\.\d+)?)')
topic_re = re.compile(r'(oig_local/[A-Za-z0-9_./-]+)')

rows = []
with source_path.open('rb') as f:
    for idx, bline in enumerate(f, 1):
        if b'oig_local/' not in bline:
            continue
        line = bline.decode('utf-8', errors='replace')
        ts_m = line_ts_re.search(line)
        topic_m = topic_re.search(line)
        if not ts_m or not topic_m:
            continue
        rows.append({
            'ts': ts_m.group(1).replace(' ', 'T') + 'Z',
            'source': f'.sisyphus/evidence/task-2-loki-sample.json:{idx}',
            'topic': topic_m.group(1),
            'payload_hash': hashlib.sha256(bline).hexdigest(),
        })

rows.sort(key=lambda r: r['ts'])

with out_path.open('w', encoding='utf-8', newline='\n') as f:
    for row in rows:
        f.write(json.dumps(row, ensure_ascii=False, separators=(',', ':')) + '\n')
